## Overview

`GrvtClient` is the main REST API client for interacting with GRVT's trading platform. It provides methods for managing accounts, transfers, deposits, withdrawals, and retrieving account history.

## Constructor

```typescript
new GrvtClient(config: IGrvtConfig)
```

Creates a new instance of the GRVT REST API client.

<ParamField path="config" type="IGrvtConfig" required>
  Configuration object for the client
  
  <Expandable title="IGrvtConfig properties">
    <ParamField path="apiKey" type="string">
      Your GRVT API key for authentication
    </ParamField>
    
    <ParamField path="apiSecret" type="string">
      Your GRVT API secret for signing operations (required for signing transfers and withdrawals)
    </ParamField>
    
    <ParamField path="tradingAccountId" type="string">
      The trading account ID to use for operations
    </ParamField>
    
    <ParamField path="env" type="EGrvtEnvironment" required>
      The environment to connect to:
      - `EGrvtEnvironment.PRODUCTION` - Production environment
      - `EGrvtEnvironment.TESTNET` - Testnet environment
      - `EGrvtEnvironment.STAGING` - Staging environment
      - `EGrvtEnvironment.DEV` - Development environment
    </ParamField>
  </Expandable>
</ParamField>

### Example

```typescript
import { GrvtClient, EGrvtEnvironment } from '@grvt/client';

const client = new GrvtClient({
  apiKey: 'your-api-key',
  apiSecret: 'your-api-secret',
  env: EGrvtEnvironment.TESTNET
});
```

## Transfer Methods

### transfer

Transfer funds between accounts. Automatically signs the transfer if `apiSecret` is configured.

```typescript
async transfer(
  request: IApiTransferRequest,
  metadata?: ITransferMetadata,
  options?: ISigningOptions
): Promise<IApiTransferResponse>
```

<ParamField path="request" type="IApiTransferRequest" required>
  The transfer request parameters
  
  <Expandable title="IApiTransferRequest properties">
    <ParamField path="from_account_id" type="string" required>
      The account ID to transfer from
    </ParamField>
    
    <ParamField path="to_account_id" type="string" required>
      The account ID to transfer to
    </ParamField>
    
    <ParamField path="currency" type="string" required>
      The currency to transfer (e.g., 'USDT', 'BTC')
    </ParamField>
    
    <ParamField path="num_tokens" type="string" required>
      The amount to transfer as a string
    </ParamField>
    
    <ParamField path="signature" type="ISignature">
      Optional pre-generated signature. If not provided and `apiSecret` is configured, the SDK will sign automatically
    </ParamField>
  </Expandable>
</ParamField>

<ParamField path="metadata" type="ITransferMetadata">
  Optional transfer metadata for tracking
  
  <Expandable title="ITransferMetadata properties">
    <ParamField path="provider" type="ETransferProvider" required>
      The transfer provider (e.g., `ETransferProvider.RHINO`)
    </ParamField>
    
    <ParamField path="direction" type="ETransferDirection" required>
      Transfer direction: `ETransferDirection.DEPOSIT` or `ETransferDirection.WITHDRAWAL`
    </ParamField>
    
    <ParamField path="chainid" type="EChain" required>
      The chain ID for the transfer
    </ParamField>
    
    <ParamField path="endpoint" type="string" required>
      The endpoint for the transfer provider
    </ParamField>
    
    <ParamField path="provider_tx_id" type="string">
      Optional transaction ID from the provider
    </ParamField>
    
    <ParamField path="provider_ref_id" type="string">
      Optional reference ID from the provider
    </ParamField>
  </Expandable>
</ParamField>

<ParamField path="options" type="ISigningOptions">
  Optional signing options
  
  <Expandable title="ISigningOptions properties">
    <ParamField path="nonce" type="number">
      Custom nonce for the signature (defaults to random)
    </ParamField>
    
    <ParamField path="expiration" type="string">
      Signature expiration timestamp (defaults to 24 hours from now)
    </ParamField>
  </Expandable>
</ParamField>

<ResponseField name="IApiTransferResponse" type="object">
  The transfer response containing the transfer ID and status
</ResponseField>

#### Example

```typescript
import { ETransferProvider, ETransferDirection, EChain } from '@grvt/client';

const transferResponse = await client.transfer(
  {
    from_account_id: 'funding-account-123',
    to_account_id: 'sub-account-456',
    currency: 'USDT',
    num_tokens: '1000'
  },
  {
    provider: ETransferProvider.RHINO,
    direction: ETransferDirection.DEPOSIT,
    chainid: EChain.ARBITRUM,
    endpoint: 'https://api.rhino.fi',
    provider_tx_id: 'rhino-tx-123'
  }
);
console.log('Transfer ID:', transferResponse.transfer_id);
```

### getTransferHistory

Retrieves transfer history for the account within a specified time range.

```typescript
async getTransferHistory(
  request: IApiTransferHistoryRequest
): Promise<IApiTransferHistoryResponse>
```

<ParamField path="request" type="IApiTransferHistoryRequest" required>
  Request parameters for transfer history
  
  <Expandable title="IApiTransferHistoryRequest properties">
    <ParamField path="start_time" type="string" required>
      Start time for the query (ISO 8601 format or Unix timestamp)
    </ParamField>
    
    <ParamField path="end_time" type="string" required>
      End time for the query (ISO 8601 format or Unix timestamp)
    </ParamField>
    
    <ParamField path="sub_account_id" type="string">
      Optional sub-account ID to filter transfers
    </ParamField>
  </Expandable>
</ParamField>

<ResponseField name="IApiTransferHistoryResponse" type="object">
  Array of transfer records with metadata
</ResponseField>

#### Example

```typescript

  start_time: '2024-01-01T00:00:00Z',
  end_time: '2024-01-31T23:59:59Z',
  sub_account_id: '123'
});
console.log('Transfers:', transferHistory);
```

## Deposit Methods

### deposit

Deposit funds to the account using L1 bridge or direct transfer for Arbitrum.

```typescript
async deposit(options: IDepositOptions): Promise<string>
```

<ParamField path="options" type="IDepositOptions" required>
  Deposit options
  
  <Expandable title="IDepositOptions properties">
    <ParamField path="privateKey" type="string" required>
      The private key to sign the deposit transaction
    </ParamField>
    
    <ParamField path="l1Sender" type="string" required>
      The L1 sender address
    </ParamField>
    
    <ParamField path="l2Receiver" type="string" required>
      The L2 receiver address on GRVT
    </ParamField>
    
    <ParamField path="amount" type="string" required>
      The amount to deposit as a string
    </ParamField>
    
    <ParamField path="chainId" type="EChain" required>
      The chain ID for the deposit
    </ParamField>
  </Expandable>
</ParamField>

<ResponseField name="string" type="string">
  The deposit transaction hash
</ResponseField>

#### Example

```typescript
import { EChain } from '@grvt/client';

const txHash = await client.deposit({
  privateKey: 'your-private-key',
  l1Sender: '0x1234...',
  l2Receiver: '0x5678...',
  amount: '1000000000', // Amount in token's smallest unit
  chainId: EChain.ETHEREUM
});
console.log('Deposit TX:', txHash);
```

### getGravityChainIDFromRhinoChain

Get the Gravity chain ID for a given Rhino chain.

```typescript
async getGravityChainIDFromRhinoChain(rhinoChain: string): Promise<EChain | null>
```

<ParamField path="rhinoChain" type="string" required>
  The Rhino chain identifier
</ParamField>

<ResponseField name="EChain | null" type="EChain | null">
  The corresponding Gravity chain ID, or null if not found
</ResponseField>

#### Environment Mapping

The method maps GRVT environments to Rhino environments:
- `PRODUCTION` → Rhino PROD
- `TESTNET` → Rhino STG
- `STAGING` / `DEV` → Rhino DEV

#### Example

```typescript
const chainId = await client.getGravityChainIDFromRhinoChain('ARBITRUM');
if (chainId) {
  console.log('Gravity Chain ID:', chainId);
}
```
