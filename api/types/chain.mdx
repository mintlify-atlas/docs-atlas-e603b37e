## Example Usage

```typescript
import { EChain } from '@grvt/sdk';

// Deposit from Ethereum mainnet
await client.deposit({
  chainId: EChain.ETHEREUM,
  currency: 'USDC',
  amount: '1000'
});

// Deposit from Arbitrum
await client.deposit({
  chainId: EChain.ARBITRUM_ONE,
  currency: 'USDC',
  amount: '500'
});

// Use testnet for development
await client.deposit({
  chainId: EChain.ETHEREUM_SEPOLIA,
  currency: 'USDC',
  amount: '100'
});

// Solana deposits
await client.deposit({
  chainId: EChain.SOLANA,
  currency: 'SOL',
  amount: '10'
});
```
